# Code
```cs
static class Program
{
    static void Main()
    {
        string season;
        Console.WriteLine("What is the month? 1-12");
        bool monthRes = int.TryParse(Console.ReadLine(), out int month);
        Console.WriteLine("What is the day? 1-31");
        bool dayRes = int.TryParse(Console.ReadLine(), out int day);
        month %= 12;
        month = (month != 0) ? month : 12;
        day %= 31;
        season = month switch
        {
            >= 3 and < 6 => (month == 3 && day < 21) ? "Winter" : "Spring",
            >= 6 and < 9 => (month == 6 && day < 21) ? "Spring" : "Summer",
            >= 9 and < 12 => (month == 9 && day < 21) ? "Summer" : "Autumn",
            12 or < 3 => (month == 12 && day < 21) ? "Autumn" : "Winter",
            _ => "",
        };
            Console.WriteLine($"On {day}-{month} it is {season}");   
    }
}
```
