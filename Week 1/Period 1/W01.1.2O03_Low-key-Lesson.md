# Overview
This exercise requires the programmer to create a multiple choice quiz about `C#` with the given template of questions.
Each correctly answered question had to be saved as a variable to be printed at the end.
For some reason, this one took a very long time to do, despite it probably not being that hard.
I just couldn't figure out how to make the multiple answer MCQ work with a function, so I made it more specific to the question in this example.

# Code
```cs
static class Program
{
    static int CheckAnswers(int points)
    {
        string answers;
        Console.WriteLine("Your first answer: ");
        string input1 = Console.ReadLine() ?? "";
        Console.WriteLine("Your second answer: ");
        string input2 = Console.ReadLine() ?? "";
        switch (input1.ToUpper())
        {
            case "A":
                answers = "D";
                break;
            case "D":
                answers = "A";
                break;
            default:
                return points;
        }
        switch (input2.ToUpper().Equals(answers))
        {
            case true:
                points++;
                break;
        }
        return points;
    }
    static int CheckAnswer(string answer, int points)
    {
        string input = Console.ReadLine() ?? "";
        switch (input.ToUpper().Equals(answer))
        {
            case true:
                points++;
                break;
        }
        return points;
    }
    static void Main()
    {
        int points = 0;
        Console.WriteLine("Answer the following MCQs:");
        // Question 1
        Console.WriteLine("""
        Which of the following is NOT a valid type in C#?
        A: bool
        B: int
        C: var
        D: string
        """);
        points = CheckAnswer("C", points);
        // Question 2
        Console.WriteLine("""
        What happens if you execute the following line C#?
        int x = 1.23;
        A: x will be 1.23
        B: x will be 1
        C: x will be 1.0
        D: you will get a compiler error
        """);
        points = CheckAnswer("D", points);
        // Question 3
        Console.WriteLine("""
        Consider the following line:
        double d = 1.23;
        What are TWO ways to convert variable d to an int?
        A: int i = (int)d;
        B: int i = int(d)
        C: int i = 0 + d
        D: int i = Convert.ToInt32(d)
        """);
        points = CheckAnswers(points);
        if (points != 3)
        {
            Console.WriteLine($"Your score: {points} out of 3.");
        }
        else
        {
            Console.WriteLine($"Your score: {points} out of 3. Well done!");   
        }
    }
}
```
