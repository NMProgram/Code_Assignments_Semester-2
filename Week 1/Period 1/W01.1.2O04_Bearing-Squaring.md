# Overview
This exercise required the programmer to generate an ASCII-art direction string based on which directions are allowed through the user's input.
Then, the user gives a bearing (angle) to go to and the program prints out whether that's a valid direction or not.
I desperately wanted to make this as efficient as possible, but I was unable to figure out how to manipulate the string to be built from the ground up,
so I just resorted to making a very long switch expression instead, because that's way easier to do.

# Code
```cs
static class Program
{
    static string Directions(bool n, bool w, bool e, bool s)
    {
        return (n, w, e, s) switch
        {
            (true, true, true, true) => 
            """
                N
                |
            W---|---E
                |
                S
            """,
            (false, true, true, true) => 
            """
            W---|---E
                |
                S
            """,
            (true, false, true, true) => 
            """
                N
                |
                |---E
                |
                S
            """,
            (true, true, false, true) => 
            """
                N
                |
            W---|
                |
                S
            """,
            (true, true, true, false) => 
            """
                N
                |
            W---|---E
            """,
            (false, false, true, true) => 
            """
                |---E
                |
                S
            """,
            (true, false, false, true) => 
            """
                N
                |
                |
                |
                S
            """,
            (true, true, false, false) => 
            """
                N
                |
            W---|
            """,
            (false, false, false, true) => 
            """
                |
                |
                S
            """,
            (true, false, false, false) => 
            """
                N
                |
                |
            """,
            (true, false, true, false) => 
            """
                N
                |
                |---E
            """,
            (false, true, true, false) => 
            """
            W---|---E
            """,
            (false, true, false, true) => 
            """
            W---|
                |
                S
            """,
            (false, true, false, false) => 
            """
            W---|
            """,
            (false, false, true, false) => 
            """
                |---E
            """,
            (false, false, false, false) => 
            """
                |
            """,
        };
    }
    static void Main()
    {
        // Directions allowed
        Console.WriteLine("For each direction, input Y/N if it is valid.");
        Console.WriteLine("North? Y/N");
        string north = Console.ReadLine() ?? "";
        bool northBool = north.Contains('Y');
        Console.WriteLine("East? Y/N");
        string east = Console.ReadLine() ?? "";
        bool eastBool = east.Contains('Y');
        Console.WriteLine("South? Y/N");
        string south = Console.ReadLine() ?? "";
        bool southBool = south.Contains('Y');
        Console.WriteLine("West? Y/N");
        string west = Console.ReadLine() ?? "";
        bool westBool = west.Contains('Y');
        string formedDirection = Directions(northBool, westBool, eastBool, southBool);
        // Angle calculation
        Console.WriteLine("Give a bearing (a number) for the direction in which you are going.");
        bool result = int.TryParse(Console.ReadLine(), out int angle);
        angle %= 360;
        angle = (angle < 0) ? angle + 360 : angle;
        string direction = angle switch
        {
            (> 315 and < 360) or (> 0 and <= 45) => "North",
            > 45 and <= 135 => "East",
            > 135 and <= 225 => "South",
            > 225 and <= 315 => "West",
            _ => "Invalid"
        };
        // Printing Direction
        Console.WriteLine("From here you can go:");
        Console.WriteLine(formedDirection + "\n");
        switch(direction)
        {
            case "North" when northBool:
                Console.WriteLine("You are going North");
                break;
            case "East" when eastBool:
                Console.WriteLine("You are going East");
                break;
            case "South" when southBool:
                Console.WriteLine("You are going South");
                break;
            case "West" when westBool:
                Console.WriteLine("You are going West");
                break;
            default:
                Console.WriteLine($"You can't go {direction}");
                break;
        }
    }
}
```
