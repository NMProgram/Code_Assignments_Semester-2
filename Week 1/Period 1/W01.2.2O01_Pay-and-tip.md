# Overview
This exercise required the programmer to make an application that asks the user for an amount to pay and proceeds to ask the user how much they'd like to pay,
until the cost is equal or less than 0. If it's less than 0, the user also has to say whether they'd like to give a tip or pay the exact amount left.

This one admittedly took longer than it should've, and it's kinda messy the way it turned out because I misunderstood what I had to do for this exercise. 
But it wasn't all too bad.

# Code
```cs
static class Program
{
    static void Loop(int cost)
    {
        int ogCost = cost;
        string menu = """
        Pay how much?
        1: 5
        2: 10
        3: 20
        4: 50
        """;
        Dictionary<int, int> options = new()
        {
            {1, 5},
            {2, 10},
            {3, 20},
            {4, 50},
        };
        while (cost > 0)
        {
            Console.WriteLine(menu);
            int choice = int.Parse(Console.ReadLine()!);
            bool choiceRes = options.TryGetValue(choice, out int payAmount);
            if (!choiceRes) { continue; }
            if (payAmount > cost)
            {
                int tip = (cost - payAmount) * -1;
                Console.WriteLine($"You paid {tip} too much. Give a tip? y/n");
                string answer = Console.ReadLine()!;
                if (answer.ToUpper().Equals("Y"))
                {
                    Console.WriteLine($"You have paid {ogCost + tip}");
                }
                else
                {
                    Console.WriteLine($"You have paid {ogCost}");
                }
                break;
            }
            else if (payAmount == cost)
            {
                Console.WriteLine($"You have paid {ogCost}");
                break;
            }
            Console.WriteLine($"{cost - payAmount} left to pay");
            cost -= payAmount;
        }
    }
    static void Main()
    {
        Console.WriteLine("What is the amount to pay?");
        bool costRes = int.TryParse(Console.ReadLine(), out int cost);
        if (!costRes)
        {
            Environment.Exit(0);
        }
        Loop(cost);
    }
}
```
